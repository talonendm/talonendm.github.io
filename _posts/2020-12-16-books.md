function setup() {
  // noLoop();
  
  var leveys = max(windowWidth * 0.69, windowHeight * 0.69);

  
  const canvas = createCanvas(leveys, 1200);
  canvas.parent('sketch-holder-jt-books')
  
  
  
  
  for (let i = 0; i < books.finished.length ; i++) {
    bo.push(new Book(i, "-", "_"));
    boo.push(new Book(i, books.finished[i].title + "  (" + books.finished[i].author + ")", books.finished[i].author));
  }
  
}

function draw() {
  background(255);
  fill(0,0,0);
//  for (let i = 0; i < books.finished.length ; i++) {
//    text(books.finished[i].title + "(" + books.finished[i].author + ")", 10, i*rivivali);  
 // }
  
  for(let i = 0;i<bo.length;i++) {
    bo[i].createParticle();
    bo[i].moveParticle();

  }
  
  
  for(let i = 0;i<boo.length;i++) {
    boo[i].createBook();
    boo[i].addSpeed(false);
    boo[i].moveParticle();

  }
  
  
  
  var j = floor(mouseY / rivivali);
  if (j >= books.finished.length ) j = books.finished.length - 1;
  if (j < 0) j=0; // see ctrl shift i and console
  fill(255,0,0);
  //text(books.finished[j].title + " (" + books.finished[j].author + ")", 10, j*rivivali); 
  
  
  
  for(let i = 0;i<boo.length;i++) {
    if (boo[i].getAuthor() == boo[j].getAuthor()) {
        boo[i].createBook(true);
        boo[i].addSpeed(true);
    }
  }
  
  boo[j].createBook(true);
  
}




// this class describes the properties of a single particle.
class Book {
// setting the co-ordinates, radius and the
// speed of a particle in both the co-ordinates axes.
  constructor(i_, teksti_, author_){
    this.x = round(random(0,width));
    this.y = rivivali*2 + i_ * rivivali;
    this.r = random(1,8);
    this.xSpeed = random(-1,1);
    this.ySpeed = 0; // random(-1,1.5);
    this.teksti = teksti_;
    this.marginaali = 10;
    this.author = author_;
  }

// creation of a particle.
  createParticle() {
    noStroke();
    fill('rgba(200,169,169,0.5)');
    circle(this.x,this.y,this.r);
  }
  
  createBook(hoveri = false) {
    noStroke();
    if (hoveri){
      fill('rgba(170,22,22,0.8)');
    } else {
      fill('rgba(150,169,169,0.8)');
    }
    
    circle(this.x,this.y,this.r);
    
    text(this.teksti, this.x,this.y);  

  }
  

  getAuthor() {
    return(this.author);
  }
  addSpeed(selected = false) {
    
    if (selected) {
      this.xSpeed = 2;
    } else {
    
    if (this.x > this.marginaali) {
      this.xSpeed = -1; }
    else if (round(this.x) == this.marginaali) this.xSpeed = 0;
    
    }
  }
  
// setting the particle in motion.
  moveParticle() {
    if(this.x < 0 || this.x > width)
      this.xSpeed*=-0.9;
    if(this.y < 0 || this.y > height)
      this.ySpeed*=-0.9;
    
    
    
    
    this.x+=this.xSpeed;
    this.y+=this.ySpeed;
  }

  
}
